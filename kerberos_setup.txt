1. Install kerberos
2. Configure kerberos configuration file
- locate krb5.ini (C:\Program Files\MIT\Kerberos\)
If not available then create under C:\windows

[libdefaults]
    default_realm = KAFKA.COM
    dns_lookup_realm = false
    dns_lookup_kdc = true

[realms]
    KAFKA.COM = {
        kdc = kerberos-server.kafka.com
        admin_server = kerberos-server.kafka.com
    }

[domain_realm]
    .kafka.com = KAFKA.COM
    kafka.com = KAFKA.COM
